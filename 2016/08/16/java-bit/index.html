public class Solution {
    public int getSum(int a, int b) {
        while (b != 0) {
            int carry = a ^ b;
            b = (a & b) << 1;
            a = carry;
        }
        return a;
    }
}

int mul(int a, int b) {
	int res = 0;
	while (b != 0) {
		if (b & 1 != 0) {

		}
		a = (a << 1);
		b = (b >> 1);
	}
	return res;
}
